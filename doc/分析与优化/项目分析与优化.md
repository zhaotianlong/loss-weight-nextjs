#### 4. **缺少错误边界**

**问题描述：**
- 没有全局错误边界
- 组件错误会导致整个应用崩溃

**优化方案：**
```typescript
// app/error-boundary.tsx
'use client';

import { Component, ReactNode } from 'react';
import { Result, Button } from 'antd';

interface Props {
  children: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: any) {
    console.error('Error caught by boundary:', error, errorInfo);
    // 可以发送到错误追踪服务（如 Sentry）
  }

  render() {
    if (this.state.hasError) {
      return (
        <Result
          status="500"
          title="出错了"
          subTitle={this.state.error?.message || '发生了未知错误'}
          extra={
            <Button type="primary" onClick={() => window.location.reload()}>
              刷新页面
            </Button>
          }
        />
      );
    }

    return this.props.children;
  }
}
```

```typescript
// hooks/useCachedRequest.ts
import { useState, useEffect, useRef } from 'react';

const cache = new Map<string, { data: any; timestamp: number }>();
const CACHE_TTL = 5 * 60 * 1000; // 5分钟

export function useCachedRequest<T>(
  key: string,
  fetcher: () => Promise<T>,
  options?: { ttl?: number }
) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const ttl = options?.ttl || CACHE_TTL;

  useEffect(() => {
    const cached = cache.get(key);
    const now = Date.now();

    if (cached && now - cached.timestamp < ttl) {
      setData(cached.data);
      setLoading(false);
      return;
    }

    fetcher().then(result => {
      cache.set(key, { data: result, timestamp: now });
      setData(result);
      setLoading(false);
    });
  }, [key, fetcher, ttl]);

  return { data, loading };
}
```

## 🔬 深入技术分析

### 阶段一：类型系统深度分析

#### 1. **类型安全漏洞：大量使用 `as any`**

**问题统计：**
- 在 `src/mocks/handlers/` 中发现 **140+ 处** `as any` 类型断言
- 主要分布在数据转换和字段访问场景

**具体问题位置：**

```typescript
// src/mocks/handlers/user.ts - 典型问题
const data = full.slice(start, start + pageSize).map((item: Employee) => ({
  userId: (item as any).empId,        // ❌ 类型断言
  id: (item as any).empId,            // ❌ 类型断言
  name: (item as any).name,           // ❌ 类型断言
  // ... 更多字段
}));
```

**根本原因：**
1. **数据结构不匹配**：`mockData` 中的数据结构（如 `empId`）与 API 返回结构（如 `userId`）不一致
2. **类型定义缺失**：`CollectionItem<K>` 类型定义过于宽泛，缺少具体字段定义
3. **映射逻辑缺失**：缺少数据转换层统一处理字段映射

**影响：**
- ❌ 失去 TypeScript 类型检查优势
- ❌ 运行时错误风险增加
- ❌ 重构困难，容易引入 bug

```typescript
// ✅ 创建通用过滤工具
// src/utils/queryBuilder.ts
export interface FilterOptions<T> {
  name?: string;
  phone?: string;
  status?: number | null;
  campId?: number;
  // 支持自定义过滤函数
  customFilters?: Array<(item: T) => boolean>;
}

export function buildFilters<T extends Record<string, any>>(
  options: FilterOptions<T>
): Array<(item: T) => boolean> {
  const filters: Array<(item: T) => boolean> = [];

  if (options.name) {
    const nameLower = options.name.toLowerCase();
    filters.push((item) =>
      item.name?.toLowerCase().includes(nameLower) ?? false
    );
  }

  if (options.phone) {
    filters.push((item) => item.phone?.includes(options.phone!) ?? false);
  }

  if (options.status !== null && options.status !== undefined) {
    filters.push((item) => item.status === options.status);
  }

  if (options.campId) {
    filters.push((item) => item.campId === options.campId);
  }

  if (options.customFilters) {
    filters.push(...options.customFilters);
  }

  return filters;
}

export function applyFilters<T>(
  items: T[],
  filters: Array<(item: T) => boolean>
): T[] {
  return filters.reduce((acc, filter) => acc.filter(filter), items);
}

// 使用示例
const filters = buildFilters<Employee>({
  name: req.url.searchParams.get('name') || undefined,
  phone: req.url.searchParams.get('phone') || undefined,
  status: req.url.searchParams.get('status')
    ? Number(req.url.searchParams.get('status'))
    : undefined,
});
const filtered = applyFilters(full, filters);
```

```typescript
export interface PaginationOptions {

  pageSize?: number;
}

export interface PaginatedResult<T> {
  data: T[];
  meta: {
    page: number;
    pageSize: number;
    total: number;
    pageCount: number;
  };
}

export function paginate<T>(
  items: T[],
  options: PaginationOptions = {}
): PaginatedResult<T> {
  const page = options.page ?? 1;
  const pageSize = options.pageSize ?? 20;
  const total = items.length;
  const start = (page - 1) * pageSize;
  const end = start + pageSize;
  const data = items.slice(start, end);

  return {
    data,
    meta: {
      page,
      pageSize,
      total,
      pageCount: Math.ceil(total / pageSize),
    },
  };
}
```

```typescript
export function useAsyncOperation<T, P extends any[]>(
  asyncFn: (...args: P) => Promise<T>
) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [data, setData] = useState<T | null>(null);
  
  const execute = useCallback(async (...args: P) => {
    setLoading(true);
    setError(null);
    try {
      const result = await asyncFn(...args);
      setData(result);
      return result;
    } catch (err) {
      const error = err instanceof Error ? err : new Error(String(err));
      setError(error);
      throw error;
    } finally {
      setLoading(false);
    }
  }, [asyncFn]);
  
  return { loading, error, data, execute };
}
```
